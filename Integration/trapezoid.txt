function area = trapezoid
    
    A = read("200600237.DAT",-1,2);
    area = 0;
    x = A(:,1);
    y = A(:,2);
    n = size(A,1)-1;
    
    for i=1:n
        
        area = area + (x(i+1)-x(i))*((y(i)+y(i+1))/2);        
    
    end
    
    
    
endfunction
